66int AnalogPort::readADC(int channel, int adcChipSelectPin)
67{
68 /* Sets default Primary ADC Address register B00000110, This is a default
69 address setting, the third LSB is set to one to start the ADC. The second
70 LSB is to set the ADC to single ended mode, the LSB is for D2 address bit,
71 for this ADC its a "Don't Care" bit. */
72 byte adcPrimaryRegister = 0b00000110;
73
74 // b00000111 Isolates the three LSB.
75 byte adcPrimaryRegisterMask = 0b00000111;
76
77 // b00001111 isolates the 4 LSB for the value returned.
78 byte adcPrimaryByteMask = 0b00001111;
79 byte adcD2Config = channel >> 2;
80
81 /* Ensures the D2 bit of the primary register corresponds to the bit as
82 determined by the channel. */
83 byte adcPrimaryConfig = adcPrimaryRegister | adcD2Config ;
84
85 /* Ensures the adc register is limited to the mask and assembles the
86 configuration byte to send to ADC. */
87 adcPrimaryConfig &= adcPrimaryRegisterMask;
88 byte adcSecondaryConfig = channel << 6;
89
90 // Disable interupts to prepare to send address data to the ADC.
91 noInterrupts();
92
93 // Take the Chip Select pin low to select the ADC.
94 digitalWrite(adcChipSelectPin, LOW);
95
96 // Send in the primary configuration address byte to the ADC.
97 SPI.transfer(adcPrimaryConfig);
98
99 // Read the primary byte, also sending in the secondary address byte.
100 byte adcPrimaryByte = SPI.transfer(adcSecondaryConfig);
101
102 // Read the secondary byte, also sending 0 as this doesn't matter.
103 byte adcSecondaryByte = SPI.transfer(0x00);
104
105 // Take the Chip Select pin high to de-select the ADC.
106 digitalWrite(adcChipSelectPin, HIGH);
107
108 // Enable interupts.
109 interrupts();
110
111 // Limits the value of the primary byte to the 4 LSB:
112 adcPrimaryByte &= adcPrimaryByteMask;
113
114 /* Shifts the 4 LSB of the primary byte to become the 4 MSB of the 12
115 bit digital value, this is then ORed to the secondary byte value that
116 holds the 8 LSB of the digital value. */
117 int digitalValue = (adcPrimaryByte << 8) | adcSecondaryByte;
118
119 // Returns the value from the function
120 return digitalValue;
121}
