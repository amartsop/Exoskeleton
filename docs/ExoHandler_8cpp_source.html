7void ExoHandler::initialize(void)
8{
9 // Intialize chip selct pins
10 pinMode(m_adc_chip_select_pin1, OUTPUT);
11 pinMode(m_adc_chip_select_pin2, OUTPUT);
12 pinMode(m_adc_test_pin, m_test_pin);
13
14 // Set the select pins high initially:
15 digitalWrite(m_adc_chip_select_pin1, HIGH);
16 digitalWrite(m_adc_chip_select_pin2, HIGH);
17
18 // Initialise SPI:
19 SPI.begin();
20 SPI.setBitOrder(MSBFIRST); // Not strictly needed but just to be sure.
21 SPI.setDataMode(SPI_MODE0); // Not strictly needed but just to be sure.
22
23 /* Set SPI clock divider to 16, therfore a 1 MhZ signal due
24 to the maximum frequency of the ADC. */
25 SPI.setClockDivider(SPI_CLOCK_DIV16);
26
27 // Initialize SerialUSB communication
28 SerialUSB.begin(m_baudrate);
29
30 // Initialize analog ports
31 initialize_analog_ports();
32
33 // Initialize sensors
34 initialize_sensors();
35}

41void ExoHandler::update_measurements(void)
42{
43 // Update analog inputs
44 for (size_t i = 0; i < analog_ports_num; i++)
45 {
46 m_analog_ports_container.port[i].update_port_values();
47 }
48
49 // Update sensors
50 for (size_t i = 0; i < sensors_num; i++)
51 {
52 // Get port id for sensor i
53 int port_id = m_sensors_container.sensor[i].get_port_id();
54
55 // Get port values for port id
56 AnalogPort::AnalogPortValues analog_values =
57 m_analog_ports_container.port[port_id - 1].get_port_values();
58
59 // Update sensor
60 m_sensors_container.sensor[i].update(analog_values);
61 }
